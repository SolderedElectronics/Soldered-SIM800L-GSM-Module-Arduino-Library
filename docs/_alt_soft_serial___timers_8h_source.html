133 // CH5 = input capture (input, pin 20)
134 // CH6 = compare a (output, pin 21)
135 // CH0 = compare b (input timeout)
136 #define CONFIG_TIMER_NOPRESCALE() \
137  FTM0_SC = 0; \
138  FTM0_CNT = 0; \
139  FTM0_MOD = 0xFFFF; \
140  FTM0_SC = FTM_SC_CLKS(1) | FTM_SC_PS(0); \
141  digitalWriteFast(21, HIGH); \
142  NVIC_SET_PRIORITY(IRQ_FTM0, 48); \
143  FTM0_C0SC = 0x18; \
144  NVIC_ENABLE_IRQ(IRQ_FTM0);
145 #define CONFIG_TIMER_PRESCALE_8() \
146  FTM0_SC = 0; \

150  digitalWriteFast(21, HIGH); \
151  NVIC_SET_PRIORITY(IRQ_FTM0, 48); \
152  FTM0_C0SC = 0x18; \
153  NVIC_ENABLE_IRQ(IRQ_FTM0);
154 #define CONFIG_TIMER_PRESCALE_128() \
155  FTM0_SC = 0; \
156  FTM0_CNT = 0; \

159  digitalWriteFast(21, HIGH); \
160  NVIC_SET_PRIORITY(IRQ_FTM0, 48); \
161  FTM0_C0SC = 0x18; \
162  NVIC_ENABLE_IRQ(IRQ_FTM0);
163 #define CONFIG_MATCH_NORMAL() (FTM0_C6SC = 0)
164 #define CONFIG_MATCH_TOGGLE() (FTM0_C6SC = (FTM0_C6SC & 0xC3) | 0x14)
165 #define CONFIG_MATCH_CLEAR() (FTM0_C6SC = (FTM0_C6SC & 0xC3) | 0x18)
166 #define CONFIG_MATCH_SET() (FTM0_C6SC = (FTM0_C6SC & 0xC3) | 0x1C)
167 #define CONFIG_CAPTURE_FALLING_EDGE() (FTM0_C5SC = (FTM0_C5SC & 0xC3) | 0x08)
168 #define CONFIG_CAPTURE_RISING_EDGE() (FTM0_C5SC = (FTM0_C5SC & 0xC3) | 0x04)
169 #define ENABLE_INT_INPUT_CAPTURE() \
170  FTM0_C5SC = 0x48; \
171  CORE_PIN20_CONFIG = PORT_PCR_MUX(4) | PORT_PCR_PE | PORT_PCR_PS
172 #define ENABLE_INT_COMPARE_A() \
173  FTM0_C6SC |= 0x40; \
174  CORE_PIN21_CONFIG = PORT_PCR_MUX(4) | PORT_PCR_DSE | PORT_PCR_SRE
175 #define ENABLE_INT_COMPARE_B() (FTM0_C0SC = 0x58)
176 #define DISABLE_INT_INPUT_CAPTURE() \
177  FTM0_C5SC &= ~0x40; \
178  CORE_PIN20_CONFIG = PORT_PCR_MUX(1) | PORT_PCR_PE | PORT_PCR_PS
179 #define DISABLE_INT_COMPARE_A() \
180  FTM0_C6SC &= ~0x40; \
181  CORE_PIN21_CONFIG = PORT_PCR_MUX(1) | PORT_PCR_DSE | PORT_PCR_SRE; \
182  digitalWriteFast(21, HIGH)
183 #define DISABLE_INT_COMPARE_B() (FTM0_C0SC &= ~0x40)
184 #define GET_TIMER_COUNT() (FTM0_CNT)
185 #define GET_INPUT_CAPTURE() (FTM0_C5V)
186 #define GET_COMPARE_A() (FTM0_C6V)
187 #define GET_COMPARE_B() (FTM0_C0V)
188 #define SET_COMPARE_A(val) (FTM0_C6V = val)
189 #define SET_COMPARE_B(val) \
190  if (FTM0_C0SC & FTM_CSC_CHF) \
191  FTM0_C0SC = 0x18; \
192  do \
193  { \
194  FTM0_C0V = (val); \
195  } while (FTM0_C0V != (val));
